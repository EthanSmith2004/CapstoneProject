\section{Doelwitte}

\subsection{Outomatisering van die Bestelproses}
\begin{itemize}
  \item Ontwikkeling van 'n mobiele toepassing waar kliënte maklik weeklikse maaltye vanaf die spyskaart kan kies en bestel.
  \item Implementering van 'n bevestigingstelsel wat outomaties kennisgewings stuur wanneer bestellings geplaas, verander of gekanselleer word.
  \item Vermindering van administratiewe tyd deur outomatiese verwerking van bestellings.
\end{itemize}

\subsection{Spyskaartbestuur en Administrasie}
\begin{itemize}
  \item Ontwikkel 'n web gebaseerde administrasiepaneel waarmee Spys spyskaarte kan byvoeg en verwyder, opdaterings kan maak en enige nodige administrasie-werk kan doen.
  \item Implementering van 'n gebruikersvriendelike spyskaartboumodule.
\end{itemize}

\subsection{Finansiële Monitering}
\begin{itemize}
  \item Bied finansiële verslaggewing sodat Akademia betaling deur student- en personeelrekeninge kan hanteer.
  \item Ontwikkeling van gedetailleerde finansiële verslae wat uitgawes per koshuis en per gebruikersgroep kan analiseer.
  \item Integrasie van 'n kredietpuntestelsel wat saldo's van studente en personeel akkuraat byhou.
\end{itemize}

\subsection{Gebruikersregistrasie en Profielbestuur}
\begin{itemize}
  \item Laat gebruikers toe om self aan te meld en hul gebruikersinligting te wysig.
  \item Sorg ook dat gebruikers onafhanklik van Spys of Akademia se huidige bestaande stelsels funksioneer.
  \item Implementeer 'n allergiebestuurstelsel wat gebruikers toelaat om voedselallergieë aan te dui en outomaties gewaarsku te word oor onveilige maaltye.
\end{itemize}

\subsection{Gebruikersvriendelikheid en Toeganklikheid}
\begin{itemize}
  \item Verseker dat beide die mobiele en web toepassings eenvoudig en toeganklik is vir gebruikers op verskillende toestelle.
  \item Ontwikkel 'n intuïtiewe koppelvlak wat minimale opleiding vereis.
  \item Implementeer responsiewe ontwerp wat op alle skermgroottes werk, van slimfone tot rekenaarskerms.
\end{itemize}

\subsection{Rekord van Bestellings}
\begin{itemize}
  \item Verminder misverstand en foutiewe bestellings deur duidelike bestel geskiedenis en bevestigings te bied.
  \item Maak bestellings-inligting maklik toeganklik vir beide gebruikers en administrateurs.
  \item Implementeer geskiedenisbesigtiging wat gebruikers toelaat om vorige bestellings te herhaal.
\end{itemize}
