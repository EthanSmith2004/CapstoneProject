\section{Omvang}

\subsection{Gebruikerskoppelvlakke}
\begin{itemize}
    \item Ontwikkeling van 'n mobiele toepassing waarmee gebruikers etes kan besigtig, bestel en hul bestellings geskiedenis kan vind.
    \item Ontwikkeling van 'n web gebaseerde platform vir Spys personeel en Akademia personeel om administratiewe take te voltooi insluitend die bestuur van spyskaarte en die hantering van finansiële rekeninge.
    \item Die mobiele toepassing sal ontwikkel word vir beide Android en iOS platforms deur gebruik te maak van 'n \ac{PWA}. 
    \item Die web toepassing sal werk op moderne webblaaiers.
\end{itemize}

\subsection{Datahantering}
\begin{itemize}
    \item Ontwikkeling van 'n databasisstruktuur vir die entiteite wat nodig sal wees vir die bestendigheid van data.
    \item Besigheidslogika vir die hantering van bestellings per dag, per gebruiker, en per spyskaart.
    \item Implementering van 'n betroubare rugsteun- en hersteloplossing.
    \item Ontwikkeling van 'n dataontledingsgereedskappe vir Spys bestuur om tendense in bestellings te identifiseer.
\end{itemize}

\subsection{Verslaggewing}
\begin{itemize}
    \item Administratiewe funksie om verslae te genereer vir rekeningdoeleindes.
    \item Administratiewe funksie om verslae te genereer vir die maak en aflewering van kos.
    \item Ontwikkeling van weeklikse en maandelikse verslae oor bestellingspatrone.
    \item Implementering van gebruikers-terugvoerverslae oor maaltydgehalte.
\end{itemize}

\subsection{Sekuriteit}
\begin{itemize}
    \item Gebruiker profiele is geïsoleerd vanaf mekaar.
    \item Administrasie funksionaliteit kan net deur gemagtigde personeel gebruik word.
    \item Databasis stoor wagwoorde as 'n hashwaarde.
    \item Voldoening aan \ac{POPIA}.
\end{itemize}

\subsection{Aspekte uitgesluit van die projekomvang}
\begin{itemize}
    \item Ad-hoc spyskaartdienste
    \item Fisiese afleweringslogistiek
    \item Inkoping van voedselvoorraad
    \item Aansluiting by huidige Akademia of Spys stelsels
    \item Ontwikkeling van betaalkaartintegrasie vir eksterne betalings
\end{itemize}
