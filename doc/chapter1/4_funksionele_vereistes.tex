\section{Funksionele Vereistes}

\subsection{Mobiele Toepassing - Gebruikersgerig}

\subsubsection{Registrasie}
\begin{description}
  \item[FREQ-USR-001] Gebruikers moet kan registreer met volledige naam, van, e-posadres en wagwoord.
  \item[FREQ-USR-002] Tydens registrasie skep gebruikers wat bestellings wil plaas 'n addisionele profiel met studentenommer/personeelnommer, kampus-ligging, koshuis (indien van toepassing), en allergie-inligting.
  \item[FREQ-USR-003] Administrasiepersoneel word deur 'n ander administrateur opgegradeer na die administrateur rol na registrasie.
\end{description}

\subsubsection{Intekening}
\begin{description}
  \item[FREQ-USR-101] Gebruikers moet kan inteken met e-posadres en wagwoord.
\end{description}

\subsubsection{Profielbestuur}
\begin{description}
  \item[FREQ-USR-201] Gebruikers moet hul profiel kan besigtig en sommige velde wysig.
  \item[FREQ-USR-202] Allergieë-inligting kan gestoor en verander word.
  \item[FREQ-USR-203] Koshuis en kampus kan verander word.
\end{description}

\subsubsection{Spyskaartbesigtiging}
\begin{description}
  \item[FREQ-MOB-001] Gebruikers moet die weeklikse spyskaart in 'n gebruikersvriendelike formaat kan besigtig.
  \item[FREQ-MOB-002] Allergie-aanduiders moet sigbaar wees by elke gereg.
  \item[FREQ-MOB-003] Voedingswaarde-inligting moet beskikbaar wees.
  \item[FREQ-MOB-004] Beelde van maaltye moet vertoon word indien beskikbaar.
\end{description}

\subsubsection{Bestellingsbeheer}
\begin{description}
  \item[FREQ-MOB-101] Bestellings moet vertoon kan word voor aflewering.
  \item[FREQ-MOB-102] Voltooide bestellings moet beskikbaar wees vir besigtiging.
  \item[FREQ-MOB-103] Bestellings moet gewysig kan word binne 'n gegewe tydsvenster gestel deur die administrateur.
  \item[FREQ-MOB-104] Gebruikers moet kansellasies kan maak binne 'n gegewe tydsvenster gestel deur die administrateur.
\end{description}

\subsubsection{Betaling en Finansies}
\begin{description}
  \item[FREQ-MOB-201] Betalings word vanaf 'n gebruikersrekening gedebiteer.
  \item[FREQ-MOB-202] Krediet word deur die finansiële administrateur gelaai.
  \item[FREQ-MOB-203] Gebruikers moet hul huidige rekeningbalans kan sien.
  \item[FREQ-MOB-204] Transaksie geskiedenis moet beskikbaar wees vir besigtiging.
\end{description}

\subsubsection{Terugvoer en Kommunikasie}
\begin{description}
  \item[FREQ-MOB-301] Terugvoer oor maaltye kan elektronies gegee word.
  \item[FREQ-MOB-302] Gradering van maaltye op 'n 5-punt skaal.
  \item[FREQ-MOB-303] Opsie om kommentaar en voorstelle te maak.
  \item[FREQ-MOB-304] Kennisgewings vir veranderinge in die spyskaart.
  \item[FREQ-MOB-305] Kennisgewings dat bestellings amper gereed is om af te haal.
\end{description}

\subsection{Web toepassing - Administratief}

\subsubsection{Spyskaartbestuur}
\begin{description}
  \item[FREQ-WEB-001] Die weeklikse "ete van die dag" spyskaart moet ingelees word.
  \item[FREQ-WEB-002] Die weeklikse spyskaart moet gewysig kan word.
  \item[FREQ-WEB-003] Allergene en voedingswaarde-inligting moet per gereg gevoeg kan word.
\end{description}

\subsubsection{Finansiële Administrasie}
\begin{description}  \item[FREQ-WEB-101] Akademia rekening krediet vir etes moet ingelees kan word vir studente en personeel.
  \item[FREQ-WEB-102] 'n Balans staat van studente en personeellede se rekeninge moet beskikbaar wees.
  \item[FREQ-WEB-103] Gedetailleerde finansiële verslae moet gegenereer kan word.
\end{description}

\subsubsection{Bestellingsbeheer en Administrasie}
\begin{description}  \item[FREQ-WEB-201] Verlede en gekanseleerde bestellings moet gesien kan word.
  \item[FREQ-WEB-202] 'n Staat vir die aantal etes wat vir 'n gegewe dag gemaak moet word moet gegenereer word.
  \item[FREQ-WEB-203] Aflewering plekke en besonderhede moet georganiseer kan word.
\end{description}

\subsubsection{Gebruiker- en Toegangsbeheer}
\begin{description}
  \item[FREQ-WEB-301] Verstelling van gebruikersrolle (kliënt, admin).  \item[FREQ-WEB-302] Toegangsregte moet toegeken en herroep kan word.
  \item[FREQ-WEB-303] Gebruikersgeskiedenis en -aktiwiteite moet gemonitor kan word.
  \item[FREQ-WEB-304] Stelsel wye kennisgewings moet aan alle gebruikers gestuur kan word.
\end{description}
